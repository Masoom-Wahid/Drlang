// To declare a variabel use 'mthwl'
// To assign it use 'hast'

// for ex: assigning word -> "hello world"
mthwl word hast "hello world";

// print using 'parto' keyword
parto word;

// '+' -> 'jama' , '-' -> manfi , '/' -> taqsim , '*' -> zarb
// for ex

mthwl a hast 10;
mthwl b hast 20;

parto a jama b;
parto a manfi b;
parto a zarb b;
parto a tqsim b;

// or use '+' to concatenate strings

parto word jama " nice";

// '==' is masawi and '!=' is nist
// 'true' is sahih and 'false' is ghalat

mthwl a hast sahih;
mthwl b hast ghalat;

parto a basha b;
parto b nabasha a;

// we also have a nil|null|none value and it is called 'hich';
// a variable is delcared 'hich' if there is no value given;

mthwl null_value;
parto null_value;
